% 广东工业大学本科毕业设计（论文）LaTeX 模板 —— 主文件
%
% 学生通常只需要按这个顺序改文件：
%   1. 学生信息.tex            —— 学号、姓名、题目、学院、专业等
%   2. 正文/摘要.tex           —— 中英文摘要与关键词
%   3. 正文/第一章.tex ~ 第六章.tex  —— 正文各章
%   4. 正文/致谢.tex           —— 致谢
%   5. 正文/附录A.tex          —— 附录（可选）
%   6. 参考文献.bib            —— 参考文献数据库
%   7. 图片/                   —— 放自己的插图
%
% 以下文件一般【不要改】：
%   - thesis.tex               本文件，只在新增章节时才需要改
%   - 封面.tex                 封面版式
%   - 样式/                    学校版式、字号、页边距、目录、代码块等
%
% 文档类选项：
%   ttf / otf / mac  字体方案：
%                    ttf = Windows 自带的 SimSun/SimHei
%                    otf = Overleaf 自带的 Fandol 字体
%                    mac = macOS 自带的 Songti SC / Heiti SC
%   twoside          双面镜像页边距（与学校 Word 模板一致）
%   cn / en          cn=编译中文主论文；en=编译外文译文/原文装订册
% ======================================================================
\documentclass[mac,twoside,cn]{样式/gdutthesis}
\usepackage{样式/mygdut}
\input{学生信息}

\begin{document}

\iftrans{%

  \input{外文译文/原文}%                 % 原文（en 模式）
